db.velov2026.aggregate([
  { $match: { status: "OPEN" } },
  { $group: {
      _id: "$commune",
      stations: { $sum: 1 },
      totalBikes: { $sum: "$available_bikes" },
      totalCapacity: { $sum: "$bike_stands" },
      electrical: { $sum: "$total_stands.availabilities.electricalBikes" }
  }},
  { $project: {
      _id: 0,
      commune: "$_id",
      stations: 1,
      totalBikes: 1,
      electrical: 1,
      fillRate: { $round: [
        { $multiply: [
          { $divide: ["$totalBikes", "$totalCapacity"] }, 100
        ]}, 1
      ]}
  }},
  { $sort: { fillRate: -1 } }
])
